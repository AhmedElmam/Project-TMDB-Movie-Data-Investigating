tmdb["genres"] = tmdb["genres"].astype(str)
def data(genres):
    data_plot = tmdb[genres].str.cat(sep = '|')
    data = pd.Series(data_plot.split('|'))
    info = data.value_counts(ascending=False)
    return info

total_keywords = data('genres')
print(total_keywords)
